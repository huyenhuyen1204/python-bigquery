Query gốc Query tối ưu

SELECT 

    cs.table_name AS country_name,

    cs.two_alpha_code,

    jt.density

FROM 

    `handy-station-308214.health.country_summary` AS cs

    JOIN 

        (SELECT

            ca.country_code,

            ROUND(cp.year_2019 / ca.year_2018, 2)AS density

        FROM

            `handy-station-308214.health.country_population` AS cp

            INNER JOIN `handy-station-308214.health.country_areas` AS ca

            ON ca.country_code = cp.country_code

        )AS jt

    ON cs.country_code = jt.country_code

ORDER BY jt.density 

DESC

 

SELECT 

    cs.table_name AS country_name,

    cs.two_alpha_code,

    ROUND(jt.year_2019 / jt.year_2018, 2) AS density

FROM 

    `handy-station-308214.health.country_summary` AS cs

    JOIN 

        (SELECT

            ca.country_code,

            cp.year_2019,

            ca.year_2018

        FROM

        `handy-station-308214.health.country_areas` AS ca

            INNER JOIN  `handy-station-308214.health.country_population` AS cp

            ON ca.country_code = cp.country_code

        )AS jt

    ON cs.country_code = jt.country_code

ORDER BY density 

DESC

Query gốc Query tối ưu

SELECT
   cs.table_name,
   cs.two_alpha_code,
   jt.average
FROM
   `handy-station-308214.health.country_summary` AS cs
INNER JOIN (
 SELECT
  cs.two_alpha_code,
  ROUND(AVG(hnp.value),2) AS average
FROM
  `handy-station-308214.health.country_summary` AS cs
  LEFT JOIN `handy-station-308214.health.health_nutrition_population` AS hnp
  ON hnp.country_code = cs.country_code
WHERE
  hnp.indicator_code = "SP.DYN.SMAM.MA"
  AND hnp.year > 2000
  AND hnp.value > 1.5
GROUP BY
cs.two_alpha_code
ORDER BY
average
) AS jt
ON cs.two_alpha_code = jt.two_alpha_code

 

SELECT
   cs.table_name,
   cs.two_alpha_code,
   ROUND(AVG(jt.average),2)
FROM
   `handy-station-308214.health.country_summary` AS cs
INNER JOIN (
 SELECT
  cs.two_alpha_code,
  hnp.value AS average
FROM
  `handy-station-308214.health.health_nutrition_population` AS hnp
  LEFT JOIN `handy-station-308214.health.country_summary` AS cs
  ON hnp.country_code = cs.country_code
WHERE
  hnp.indicator_code = "SP.DYN.SMAM.MA"
  AND hnp.year > 2000
  AND hnp.value > 1.5
ORDER BY
average
) AS jt
ON cs.two_alpha_code = jt.two_alpha_code
GROUP BY
cs.two_alpha_code, cs.table_name
